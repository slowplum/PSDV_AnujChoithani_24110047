    nums.append(i)

    if i%2 == 0:
        sum_even += i
    else:
        product *= i

std_div = (sum(sum_square_for_std_dev)/(len(nums) - 1))**0.5

print(sum_even)
print(product)
print(mean)
print(median)
print(std_div)

def lowerdown(i):
    return i.lower()
f = open('wordcounter.txt',"r")
counter = {}
r = f.readlines()
lines = []
for i in r:
    lines.append(i.rstrip())

for i in lines:
    words = list(map(lowerdown,i.split(' ')))
    for i in words:
        if i in counter:
            counter[i] += 1
        else:
            counter[i] = 1
print(max(counter.values()))
for i in counter:
    if counter[i] == max(counter.values()):
        print(i)
